51  /**
52   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
53   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
54   * to the <strong>"SecureChat"</strong> example in the distribution or the web
55   * site for the detailed usage.
56   *
57   * <h3>Beginning the handshake</h3>
58   * <p>
59   * You must make sure not to write a message while the
60   * {@linkplain #handshake() handshake} is in progress unless you are
61   * renegotiating.  You will be notified by the {@link ChannelFuture} which is
62   * returned by the {@link #handshake()} method when the handshake
63   * process succeeds or fails.
64   *
65   * <h3>Handshake</h3>
66   * <p>
67   * The handshake will be automaticly issued for you once the {@link Channel} is active and
68   * {@link SSLEngine#getUseClientMode()} returns {@code true}.
69   * So no need to bother with it by your self.
70   *
71   * <h3>Closing the session</h3>
72   * <p>
73   * To close the SSL session, the {@link #close()} method should be
74   * called to send the {@code close_notify} message to the remote peer.  One
75   * exception is when you close the {@link Channel} - {@link SslHandler}
76   * intercepts the close request and send the {@code close_notify} message
77   * before the channel closure automatically.  Once the SSL session is closed,
78   * it is not reusable, and consequently you should create a new
79   * {@link SslHandler} with a new {@link SSLEngine} as explained in the
80   * following section.
81   *
82   * <h3>Restarting the session</h3>
83   * <p>
84   * To restart the SSL session, you must remove the existing closed
85   * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
86   * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
87   * and start the handshake process as described in the first section.
88   *
89   * <h3>Implementing StartTLS</h3>
90   * <p>
91   * <a href="http://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
92   * communication pattern that secures the wire in the middle of the plaintext
93   * connection.  Please note that it is different from SSL &middot; TLS, that
94   * secures the wire from the beginning of the connection.  Typically, StartTLS
95   * is composed of three steps:
96   * <ol>
97   * <li>Client sends a StartTLS request to server.</li>
98   * <li>Server sends a StartTLS response to client.</li>
99   * <li>Client begins SSL handshake.</li>
100  * </ol>
101  * If you implement a server, you need to:
102  * <ol>
103  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
104  *     to {@code true},</li>
105  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
106  * <li>write a StartTLS response.</li>
107  * </ol>
108  * Please note that you must insert {@link SslHandler} <em>before</em> sending
109  * the StartTLS response.  Otherwise the client can send begin SSL handshake
110  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
111  * data corruption.
112  * <p>
113  * The client-side implementation is much simpler.
114  * <ol>
115  * <li>Write a StartTLS request,</li>
116  * <li>wait for the StartTLS response,</li>
117  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
118  *     to {@code false},</li>
119  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
120  * <li>Initiate SSL handshake by calling {@link SslHandler#handshake()}.</li>
121  * </ol>
122  *
123  * <h3>Known issues</h3>
124  * <p>
125  * Because of a known issue with the current implementation of the SslEngine that comes
126  * with Java it may be possible that you see blocked IO-Threads while a full GC is done.
127  * <p>
128  * So if you are affected you can workaround this problem by adjust the cache settings
129  * like shown below:
130  *
131  * <pre>
132  *     SslContext context = ...;
133  *     context.getServerSessionContext().setSessionCacheSize(someSaneSize);
134  *     context.getServerSessionContext().setSessionTime(someSameTimeout);
135  * </pre>
136  * <p>
137  * What values to use here depends on the nature of your application and should be set
138  * based on monitoring and debugging of it.
139  * For more details see
140  * <a href="https://github.com/netty/netty/issues/832">#832</a> in our issue tracker.
141  */

282     /**
283      * Starts an SSL / TLS handshake for the specified channel.
284      *
285      * @return a {@link ChannelPromise} which is notified when the handshake
286      *         succeeds or fails.
287      */
288     public ChannelFuture handshake(final ChannelPromise promise) {
289         final ChannelHandlerContext ctx = this.ctx;
290 
291         final ScheduledFuture<?> timeoutFuture;
292         if (handshakeTimeoutMillis > 0) {
293             timeoutFuture = ctx.executor().schedule(new Runnable() {
294                 @Override
295                 public void run() {
296                     if (promise.isDone()) {
297                         return;
298                     }
299 
300                     SSLException e = new SSLException("handshake timed out");
301                     if (promise.tryFailure(e)) {
302                         ctx.fireExceptionCaught(e);
303                         ctx.close();
304                     }
305                 }
306             }, handshakeTimeoutMillis, TimeUnit.MILLISECONDS);
307         } else {
308             timeoutFuture = null;
309         }
310 
311         ctx.executor().execute(new Runnable() {
312             @Override
313             public void run() {
314                 try {
315                     if (timeoutFuture != null) {
316                         timeoutFuture.cancel(false);
317                     }
318                     engine.beginHandshake();
319                     handshakePromises.add(promise);
320                     flush0(ctx, ctx.newPromise(), true);
321                 } catch (Exception e) {
322                     if (promise.tryFailure(e)) {
323                         ctx.fireExceptionCaught(e);
324                         ctx.close();
325                     }
326                 }
327             }
328         });
329 
330         return promise;
331     }

346         final ChannelHandlerContext ctx = this.ctx;
347         ctx.executor().execute(new Runnable() {
348             @Override
349             public void run() {
350                 engine.closeOutbound();
351                 future.addListener(closeNotifyWriteListener);
352                 try {
353                     flush(ctx, future);
354                 } catch (Exception e) {
355                     if (!future.tryFailure(e)) {
356                         logger.warn("flush() raised a masked exception.", e);
357                     }
358                 }
359             }
360         });
361

363     }

429     private void flush0(ChannelHandlerContext ctx, ChannelPromise promise, boolean internal) throws Exception {
430         final ByteBuf in = ctx.outboundByteBuffer();
431         final ByteBuf out = ctx.nextOutboundByteBuffer();
432 
433         // Do not encrypt the first write request if this handler is
434         // created with startTLS flag turned on.
435         if (!internal && startTls && !sentFirstMessage) {
436             sentFirstMessage = true;
437             out.writeBytes(in);
438             ctx.flush(promise);
439             return;
440         }
441 
442         if (ctx.executor() == ctx.channel().eventLoop()) {
443             flushFutureNotifier.add(promise, in.readableBytes());
444         } else {
445             synchronized (flushFutureNotifier) {
446                 flushFutureNotifier.add(promise, in.readableBytes());
447             }
448         }
449 
450         boolean unwrapLater = false;
451         int bytesConsumed = 0;
452         try {
453             for (;;) {
454                 SSLEngineResult result = wrap(engine, in, out);
455                 bytesConsumed += result.bytesConsumed();
456                 if (result.getStatus() == Status.CLOSED) {
457                     // SSLEngine has been closed already.
458                     // Any further write attempts should be denied.
459                     if (in.isReadable()) {
460                         in.clear();
461                         SSLException e = new SSLException("SSLEngine already closed");
462                         promise.setFailure(e);
463                         ctx.fireExceptionCaught(e);
464                         flush0(ctx, bytesConsumed, e);
465                         bytesConsumed = 0;
466                     }
467                     break;
468                 } else {
469                     switch (result.getHandshakeStatus()) {
470                     case NEED_WRAP:
471                         ctx.flush();
472                         continue;
473                     case NEED_UNWRAP:
474                         if (ctx.inboundByteBuffer().isReadable()) {
475                             unwrapLater = true;
476                         }
477                         break;
478                     case NEED_TASK:
479                         runDelegatedTasks();
480                         continue;
481                     case FINISHED:
482                         setHandshakeSuccess();
483                         continue;
484                     case NOT_HANDSHAKING:
485                         break;
486                     default:
487                         throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
488                     }
489 
490                     if (result.bytesConsumed() == 0 && result.bytesProduced() == 0) {
491                         break;
492                     }
493                 }
494             }
495 
496             if (unwrapLater) {
497                 inboundBufferUpdated(ctx);
498             }
499         } catch (SSLException e) {
500             setHandshakeFailure(e);
501             throw e;
502         } finally {
503             flush0(ctx, bytesConsumed);
504         }
505     }

507     private void flush0(final ChannelHandlerContext ctx, final int bytesConsumed) {
508         ctx.flush(ctx.newPromise().addListener(new ChannelFutureListener() {
509             @Override
510             public void operationComplete(ChannelFuture future) throws Exception {
511                 if (ctx.executor() == ctx.channel().eventLoop()) {
512                     notifyFlushFutures(bytesConsumed, future);
513                 } else {
514                     synchronized (flushFutureNotifier) {
515                         notifyFlushFutures(bytesConsumed, future);
516                     }
517                 }
518             }
519 
520             private void notifyFlushFutures(final int bytesConsumed, ChannelFuture future) {
521                 if (future.isSuccess()) {
522                     flushFutureNotifier.increaseWriteCounter(bytesConsumed);
523                     flushFutureNotifier.notifyFlushFutures();
524                 } else {
525                     flushFutureNotifier.notifyFlushFutures(future.cause());
526                 }
527             }
528         }));
529     }
530 
531     private void flush0(final ChannelHandlerContext ctx, final int bytesConsumed, final Throwable cause) {
532         ChannelFuture flushFuture = ctx.flush(ctx.newPromise().addListener(new ChannelFutureListener() {
533             @Override
534             public void operationComplete(ChannelFuture future) throws Exception {
535                 if (ctx.executor() == ctx.channel().eventLoop()) {
536                     notifyFlushFutures(bytesConsumed, cause, future);
537                 } else {
538                     synchronized (flushFutureNotifier) {
539                         notifyFlushFutures(bytesConsumed, cause, future);
540                     }
541                 }
542             }
543 
544             private void notifyFlushFutures(int bytesConsumed, Throwable cause, ChannelFuture future) {
545                 flushFutureNotifier.increaseWriteCounter(bytesConsumed);
546                 if (future.isSuccess()) {
547                     flushFutureNotifier.notifyFlushFutures(cause);
548                 } else {
549                     flushFutureNotifier.notifyFlushFutures(cause, future.cause());
550                 }
551             }
552         }));
553 
554         safeClose(ctx, flushFuture, ctx.newPromise());
555     }

557     private static SSLEngineResult wrap(SSLEngine engine, ByteBuf in, ByteBuf out) throws SSLException {
558         ByteBuffer in0 = in.nioBuffer();
559         for (;;) {
560             ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
561             SSLEngineResult result = engine.wrap(in0, out0);
562             in.skipBytes(result.bytesConsumed());
563             out.writerIndex(out.writerIndex() + result.bytesProduced());
564             if (result.getStatus() == Status.BUFFER_OVERFLOW) {
565                 out.ensureWritable(engine.getSession().getPacketBufferSize());
566             } else {
567                 return result;
568             }
569         }
570     }

585     @Override
586     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
587         if (ignoreException(cause)) {
588             // It is safe to ignore the 'connection reset by peer' or
589             // 'broken pipe' error after sending close_notify.
590             if (logger.isDebugEnabled()) {
591                 logger.debug(
592                         "Swallowing a harmless 'connection reset by peer / broken pipe' error that occurred " +
593                         "while writing close_notify in response to the peer's close_notify", cause);
594             }
595 
596             // Close the connection explicitly just in case the transport
597             // did not close the connection automatically.
598             if (ctx.channel().isActive()) {
599                 ctx.close();
600             }
601         } else {
602             ctx.fireExceptionCaught(cause);
603         }
604     }

606     /**
607      * Checks if the given {@link Throwable} can be ignore and just "swallowed"
608      *
609      * When an ssl connection is closed a close_notify message is sent.
610      * After that the peer also sends close_notify however, it's not mandatory to receive
611      * the close_notify. The party who sent the initial close_notify can close the connection immediately
612      * then the peer will get connection reset error.
613      *
614      */
615     private boolean ignoreException(Throwable t) {
616         if (!(t instanceof SSLException) && t instanceof IOException && sslCloseFuture.isDone()) {
617             String message = String.valueOf(t.getMessage()).toLowerCase();
618 
619             // first try to match connection reset / broke peer based on the regex. This is the fastest way
620             // but may fail on different jdk impls or OS's
621             if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
622                 return true;
623             }
624 
625             // Inspect the StackTraceElements to see if it was a connection reset / broken pipe or not
626             StackTraceElement[] elements = t.getStackTrace();
627             for (StackTraceElement element: elements) {
628                 String classname = element.getClassName();
629                 String methodname = element.getMethodName();
630 
631                 // skip all classes that belong to the io.netty package
632                 if (classname.startsWith("io.netty.")) {
633                     continue;
634                 }
635 
636                 // check if the method name is read if not skip it
637                 if (!"read".equals(methodname)) {
638                     continue;
639                 }
640 
641                 // This will also match against SocketInputStream which is used by openjdk 7 and maybe
642                 // also others
643                 if (IGNORABLE_CLASS_IN_STACK.matcher(classname).matches()) {
644                     return true;
645                 }
646 
647                 try {
648                     // No match by now.. Try to load the class via classloader and inspect it.
649                     // This is mainly done as other JDK implementations may differ in name of
650                     // the impl.
651                     Class<?> clazz = getClass().getClassLoader().loadClass(classname);
652 
653                     if (SocketChannel.class.isAssignableFrom(clazz)
654                             || DatagramChannel.class.isAssignableFrom(clazz)) {
655                         return true;
656                     }
657 
658                     // also match against SctpChannel via String matching as it may not present.
659                     if (PlatformDependent.javaVersion() >= 7
660                             && "com.sun.nio.sctp.SctpChannel".equals(clazz.getSuperclass().getName())) {
661                         return true;
662                     }
663                 } catch (ClassNotFoundException e) {
664                     // This should not happen just ignore
665                 }
666             }
667         }
668 
669         return false;
670     }

691      *
692      * @param   buffer
693      *                  The {@link ByteBuf} to read from. Be aware that it must have at least 5 bytes to read,
694      *                  otherwise it will throw an {@link IllegalArgumentException}.
695      * @return length
696      *                  The length of the encrypted packet that is included in the buffer. This will
697      *                  return {@code -1} if the given {@link ByteBuf} is not encrypted at all.
698      * @throws IllegalArgumentException
699      *                  Is thrown if the given {@link ByteBuf} has not at least 5 bytes to read.
700      */
701     private static int getEncryptedPacketLength(ByteBuf buffer) {
702         if (buffer.readableBytes() < 5) {
703             throw new IllegalArgumentException("buffer must have at least 5 readable bytes");
704         }
705 
706         int packetLength = 0;
707 
708         // SSLv3 or TLS - Check ContentType
709         boolean tls;
710         switch (buffer.getUnsignedByte(buffer.readerIndex())) {
711         case 20:  // change_cipher_spec
712         case 21:  // alert
713         case 22:  // handshake
714         case 23:  // application_data
715             tls = true;
716             break;
717         default:
718             // SSLv2 or bad data
719             tls = false;
720         }
721 
722         if (tls) {
723             // SSLv3 or TLS - Check ProtocolVersion
724             int majorVersion = buffer.getUnsignedByte(buffer.readerIndex() + 1);
725             if (majorVersion == 3) {
726                 // SSLv3 or TLS
727                 packetLength = (getShort(buffer, buffer.readerIndex() + 3) & 0xFFFF) + 5;
728                 if (packetLength <= 5) {
729                     // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
730                     tls = false;
731                 }
732             } else {
733                 // Neither SSLv3 or TLSv1 (i.e. SSLv2 or bad data)
734                 tls = false;
735             }
736         }
737 
738         if (!tls) {
739             // SSLv2 or bad data - Check the version
740             boolean sslv2 = true;
741             int headerLength = (buffer.getUnsignedByte(
742                     buffer.readerIndex()) & 0x80) != 0 ? 2 : 3;
743             int majorVersion = buffer.getUnsignedByte(
744                     buffer.readerIndex() + headerLength + 1);
745             if (majorVersion == 2 || majorVersion == 3) {
746                 // SSLv2
747                 if (headerLength == 2) {
748                     packetLength = (getShort(buffer, buffer.readerIndex()) & 0x7FFF) + 2;
749                 } else {
750                     packetLength = (getShort(buffer, buffer.readerIndex()) & 0x3FFF) + 3;
751                 }
752                 if (packetLength <= headerLength) {
753                     sslv2 = false;
754                 }
755             } else {
756                 sslv2 = false;
757             }
758 
759             if (!sslv2) {
760                 return -1;
761             }
762         }
763         return packetLength;
764     }

766     @Override
767     public void inboundBufferUpdated(final ChannelHandlerContext ctx) throws Exception {
768         final ByteBuf in = ctx.inboundByteBuffer();
769 
770         if (in.readableBytes() < 5) {
771             return;
772         }
773 
774         int packetLength = getEncryptedPacketLength(in);
775 
776         if (packetLength == -1) {
777             // Bad data - discard the buffer and raise an exception.
778             NotSslRecordException e = new NotSslRecordException(
779                     "not an SSL/TLS record: " + BufUtil.hexDump(in));
780             in.skipBytes(in.readableBytes());
781             ctx.fireExceptionCaught(e);
782             setHandshakeFailure(e);
783             return;
784         }
785 
786         assert packetLength > 0;
787 
788         final ByteBuf out = ctx.nextInboundByteBuffer();
789 
790         boolean wrapLater = false;
791         int bytesProduced = 0;
792         try {
793             loop:
794             for (;;) {
795                 SSLEngineResult result = unwrap(engine, in, out);
796                 bytesProduced += result.bytesProduced();
797 
798                 switch (result.getStatus()) {
799                 case CLOSED:
800                     // notify about the CLOSED state of the SSLEngine. See #137
801                     sslCloseFuture.setClosed();
802                     break;
803                 case BUFFER_UNDERFLOW:
804                     break loop;
805                 }
806 
807                 switch (result.getHandshakeStatus()) {
808                 case NEED_UNWRAP:
809                     break;
810                 case NEED_WRAP:
811                     wrapLater = true;
812                     break;
813                 case NEED_TASK:
814                     runDelegatedTasks();
815                     break;
816                 case FINISHED:
817                     setHandshakeSuccess();
818                     wrapLater = true;
819                     continue;
820                 case NOT_HANDSHAKING:
821                     break;
822                 default:
823                     throw new IllegalStateException(
824                             "Unknown handshake status: " + result.getHandshakeStatus());
825                 }
826 
827                 if (result.bytesConsumed() == 0 && result.bytesProduced() == 0) {
828                     break;
829                 }
830             }
831 
832             if (wrapLater) {
833                 flush0(ctx, ctx.newPromise(), true);
834             }
835         } catch (SSLException e) {
836             setHandshakeFailure(e);
837             throw e;
838         } finally {
839             if (bytesProduced > 0) {
840                 ctx.fireInboundBufferUpdated();
841             }
842         }
843     }

845     /**
846      * Reads a big-endian short integer from the buffer.  Please note that we do not use
847      * {@link ByteBuf#getShort(int)} because it might be a little-endian buffer.
848      */
849     private static short getShort(ByteBuf buf, int offset) {
850         return (short) (buf.getByte(offset) << 8 | buf.getByte(offset + 1) & 0xFF);
851     }

853     private static SSLEngineResult unwrap(SSLEngine engine, ByteBuf in, ByteBuf out) throws SSLException {
854         ByteBuffer in0 = in.nioBuffer();
855         for (;;) {
856             ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
857             SSLEngineResult result = engine.unwrap(in0, out0);
858             in.skipBytes(result.bytesConsumed());
859             out.writerIndex(out.writerIndex() + result.bytesProduced());
860             switch (result.getStatus()) {
861             case BUFFER_OVERFLOW:
862                 out.ensureWritable(engine.getSession().getApplicationBufferSize());
863                 break;
864             default:
865                 return result;
866             }
867         }
868     }

894     /**
895      * Notify all the handshake futures about the failure during the handshake.
896      */
897     private void setHandshakeFailure(Throwable cause) {
898 
899         // Release all resources such as internal buffers that SSLEngine
900         // is managing.
901         engine.closeOutbound();
902 
903         final boolean disconnected = cause == null || cause instanceof ClosedChannelException;
904         try {
905             engine.closeInbound();
906         } catch (SSLException e) {
907             if (!disconnected) {
908                 logger.warn("SSLEngine.closeInbound() raised an exception after a handshake failure.", e);
909             } else if (!closeNotifyWriteListener.done) {
910                 logger.warn("SSLEngine.closeInbound() raised an exception due to closed connection.", e);
911             } else {
912                 // cause == null && sentCloseNotify
913                 // closeInbound() will raise an exception with bogus truncation attack warning.
914             }
915         }
916 
917         if (!handshakePromises.isEmpty()) {
918             if (cause == null) {
919                 cause = new ClosedChannelException();
920             }
921 
922             for (;;) {
923                 ChannelPromise p = handshakePromises.poll();
924                 if (p == null) {
925                     break;
926                 }
927                 p.setFailure(cause);
928             }
929         }
930 
931         flush0(ctx, 0, cause);
932     }

991     private void safeClose(
992             final ChannelHandlerContext ctx, ChannelFuture flushFuture,
993             final ChannelPromise promise) {
994         if (!ctx.channel().isActive()) {
995             ctx.close(promise);
996             return;
997         }
998 
999         final ScheduledFuture<?> timeoutFuture;
1000         if (closeNotifyTimeoutMillis > 0) {
1001             // Force-close the connection if close_notify is not fully sent in time.
1002             timeoutFuture = ctx.executor().schedule(new Runnable() {
1003                 @Override
1004                 public void run() {
1005                     logger.warn(
1006                             ctx.channel() + " last write attempt timed out." +
1007                                             " Force-closing the connection.");
1008                     ctx.close(promise);
1009                 }
1010             }, closeNotifyTimeoutMillis, TimeUnit.MILLISECONDS);
1011         } else {
1012             timeoutFuture = null;
1013         }
1014 
1015         // Close the connection if close_notify is sent in time.
1016         flushFuture.addListener(new ChannelFutureListener() {
1017             @Override
1018             public void operationComplete(ChannelFuture f)
1019                     throws Exception {
1020                 if (timeoutFuture != null) {
1021                     timeoutFuture.cancel(false);
1022                 }
1023                 if (ctx.channel().isActive()) {
1024                     ctx.close(promise);
1025                 }
1026             }
1027         });
1028     }
